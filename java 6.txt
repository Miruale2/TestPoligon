import java.util.Scanner;

public class Main {

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int n = scanner.nextInt();
        int n1 = scanner.nextInt();
        int summ = 0;
        int k = 0;
        while (n1 != 0) {
            summ = n ^ n1;
            k = (n & n1) << 1;
            n = summ;
            n1 = k;
        }
        System.out.println(n);
    }
}
